hol_update_defense_ger_casualties = {
	if = {
		limit = {
			country_exists = GER
			OR = {
				has_war_with = GER
				has_country_flag = HOL_defense_missions_active
			}
		}
		GER = {
			set_variable = { hol_cas_search = 0 }
			while_loop_effect = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands > hol_cas_search
					}
				}
				add_to_variable = { hol_cas_search = 100 }
			}
			if = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands < hol_cas_search
					}
				}
				subtract_from_variable = { hol_cas_search = 100 }
			}
			while_loop_effect = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands > hol_cas_search
					}
				}
				add_to_variable = { hol_cas_search = 10 }
			}
			if = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands < hol_cas_search
					}
				}
				subtract_from_variable = { hol_cas_search = 10 }
			}
			while_loop_effect = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands > hol_cas_search
					}
				}
				add_to_variable = { hol_cas_search = 1 }
			}
			if = {
				limit = {
					casualties_inflicted_by = {
						opponent = HOL
						thousands < hol_cas_search
					}
				}
				subtract_from_variable = { hol_cas_search = 1 }
			}
			set_variable = { global.HOL_defense_ger_cas_temp = hol_cas_search }
		}
		set_variable = { HOL_defense_ger_casualties = global.HOL_defense_ger_cas_temp }
		multiply_variable = { HOL_defense_ger_casualties = 1000 }
	}
}
