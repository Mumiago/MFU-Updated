update_academy_weekly = {
	# Called from on_weekly, which fires once per country (ROOT = that country).
	# Operate on ROOT directly. Do NOT wrap in every_country: that would make every
	# country's weekly tick add the increment to every other country, multiplying it
	# by the (constantly changing) number of living countries.
	if = {
		limit = { has_idea = political_loyalty_spirit }
		add_to_variable = { academy_weekly_pp = 0.002 }
	}
	if = {
		limit = { has_idea = professional_officer_corps_spirit }
		add_to_variable = { academy_special_forces_capacity = 0.35 }
	}
	if = {
		limit = { has_idea = relief_of_command_spirit }
		add_to_variable = { academy_experience_gain = 0.0015 }
	}
}
