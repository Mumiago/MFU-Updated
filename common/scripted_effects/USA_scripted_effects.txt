		}

			id = USA_power_balance

		}
	}

			AND = {
				check_variable = {
					USA_crisis_level < 60 
				}

					USA_crisis_level > 49
				}
			}
		}
		add_power_balance_value = {
			id = USA_power_balance
			value = 0.05
		}
	}

		limit = {
			AND = {
				check_variable = {
					USA_crisis_level < 70
				}
				check_variable = {
					USA_crisis_level > 59
				}
			}
		}
	
	}

			AND = {

					USA_crisis_level < 80

USA_decrease_crisis_level_by_10 = {
	if = {
		limit = {
			check_variable = { USA_crisis_level > 10 }
		}
		subtract_from_variable = { USA_crisis_level = 10 }
	}
	else_if = {
		limit = {
			check_variable = { USA_crisis_level < 10 }
		}
		set_variable = { USA_crisis_level = 0 }
	}
}
