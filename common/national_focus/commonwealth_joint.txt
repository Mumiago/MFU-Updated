joint_focus = {
	id = COMMON_commonwealth_industrial_coordination
	icon = GFX_AST_allied_works

	prerequisite = { focus = COMMON_strengthen_commonwealth_ties }

	relative_position_id = COMMON_strengthen_commonwealth_ties
	x = 0
	y = 1

	cost = 5

	allow_branch = {
		OR = {
			tag = ENG
			tag = SAF
			tag = CAN
			tag = RAJ
			tag = AST
			AND = { tag = FRA has_completed_focus = FRA_refus_absurde }
			AND = { tag = POL has_completed_focus = POL_poland_is_not_yet_lost has_capitulated = yes }
		}
	}

	cancel_if_invalid = yes
	continue_if_invalid = no
	available_if_capitulated = yes

	ai_will_do = { factor = 1 }
	available = {
		OR = {
			ENG = { has_completed_focus = ENG_appeasement }
			ENG = { has_completed_focus = ENG_containment }
		}
	}

	completion_reward = {
		custom_effect_tooltip = COMMON_commonwealth_industrial_coordination_tt
	}

	completion_reward_joint_originator = {
		if = {
			limit = { ENG = { has_completed_focus = ENG_appeasement } }
			every_country = {
				limit = { OR = { tag = ENG tag = SAF tag = CAN tag = RAJ tag = AST } }
				capital_scope = {
					add_building_construction = {
						type = industrial_complex
						level = 1
						instant_build = yes
					}
				}
			}
		}
		else_if = {
			limit = { ENG = { has_completed_focus = ENG_containment } }
			every_country = {
				limit = { OR = { tag = ENG tag = SAF tag = CAN tag = RAJ tag = AST } }
				army_experience = 15
				capital_scope = {
					add_building_construction = {
						type = arms_factory
						level = 1
						instant_build = yes
					}
				}
			}
		}
	}

	completion_reward_joint_member = { }
}

joint_focus = {
	id = COMMON_commonwealth_industrial_burden_sharing
	icon = GFX_goal_CAN_depression_recovery

	prerequisite = { focus = COMMON_commonwealth_industrial_coordination }

	# Two focuses under Industrial Coordination.
	relative_position_id = COMMON_commonwealth_industrial_coordination
	x = 0
	y = 2

	cost = 3

	allow_branch = {
		OR = {
			tag = ENG
			tag = SAF
			tag = CAN
			tag = RAJ
			tag = AST
			AND = { tag = FRA has_completed_focus = FRA_refus_absurde }
			AND = { tag = POL has_completed_focus = POL_poland_is_not_yet_lost has_capitulated = yes }
		}
	}

	cancel_if_invalid = yes
	continue_if_invalid = no
	available_if_capitulated = yes

	ai_will_do = { factor = 1 }
	available = { }

	# Cost paid by the focus taker.
	completion_reward_joint_originator = {
		# Remove 1 civilian factory from each of 2 random core states (owned & controlled).
		random_owned_controlled_state = {
			limit = {
				is_core_of = ROOT
				industrial_complex > 0
			}
			save_event_target_as = COMMON_removed_civ_state_1
			remove_building = { type = industrial_complex level = 1 }
		}
		random_owned_controlled_state = {
			limit = {
				is_core_of = ROOT
				industrial_complex > 0
				NOT = { state = event_target:COMMON_removed_civ_state_1 }
			}
			remove_building = { type = industrial_complex level = 1 }
		}
	}

	# Reward for each other member.
	completion_reward_joint_member = {
		add_offsite_building = { type = industrial_complex level = 1 }
	}
}
