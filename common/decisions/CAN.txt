CAN_parliament_support = {
	CAN_shadow_factories = {
		icon = generic_political_discourse
		priority = 100
		allowed = { tag = CAN } 

		available = {
			has_completed_focus = CAN_shadow_factories
			NOT = {
				check_variable = {
					shadow_factories_CAN > 9
				}
			}

        }
		activation = {
			has_completed_focus = CAN_shadow_factories
			NOT = {
				check_variable = {
					shadow_factories_CAN > 9
				}
			}
		}
		
		selectable_mission = no
		days_mission_timeout = 50
		is_good = no
		fire_only_once = no

		days_remove = 40


		remove_effect = {
			add_to_variable = {
				var = shadow_factories_CAN
				value = 1
			}
			if = {
				limit = {
					check_variable = {
						shadow_factories_CAN < 5
					}
					
				}
				add_to_variable = {
					var = counter_CAN_1
					value = 1
				}
			}
			else_if = {
				limit = {
					check_variable = {
						shadow_factories_CAN > 5
					}
					
				}
				add_to_variable = {
					var = counter_CAN_2
					value = 1
				}
			}
			custom_effect_tooltip = add_shadow_factory
		}
		
	}
}
