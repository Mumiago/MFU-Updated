			}
		}

		visible = {
			has_completed_focus = HOL_move_industry_west

			}
		}

		is_good = yes
		fire_only_once = no
		days_mission_timeout = 25

		timeout_effect = {
			random_owned_controlled_state = {
				limit = {
					is_core_of = INS
					OR = {
						arms_factory > 0
						industrial_complex > 0
					}
				}
				if = {
					limit = {
						arms_factory > 0
					}
					remove_building = {
						type = arms_factory
						level = 1
					}
					ROOT = {
						add_offsite_building = { type = arms_factory level = 1 }
					}
				}
				else_if = {
					limit = {
						industrial_complex > 0
					}
					remove_building = {
						type = industrial_complex
						level = 1
					}
					ROOT = {
						add_offsite_building = { type = industrial_complex level = 1 }
					}
				}
			}
			hidden_effect = {
				if = {
					limit = {
						any_owned_state = {
							is_core_of = INS
							is_fully_controlled_by = ROOT
							OR = {
								arms_factory > 0
