# Scripted triggers for ENG–RAJ joint imperial defense branch (ENG_RAJ_burma_joint.txt)

ENG_RAJ_burma_joint_political_trigger = {
	OR = {
		AND = {
			original_tag = ENG
			RAJ = {
				exists = yes
				OR = {
					is_subject_of = ENG
					is_in_faction_with = ENG
				}
			}
			NOT = {
				has_war_with = RAJ
			}
		}
		AND = {
			original_tag = RAJ
			ENG = {
				exists = yes
			}
			OR = {
				is_subject_of = ENG
				is_in_faction_with = ENG
			}
			NOT = {
				has_war_with = ENG
			}
		}
	}
}

# UK must have completed the wartime cabinet / US arms focuses (required for the old non-joint entry).
ENG_RAJ_burma_joint_eng_war_leadership_trigger = {
	ENG = {
		has_completed_focus = ENG_churchill_war_minsistry
		has_completed_focus = ENG_arms_purchases_in_the_us
	}
}

ENG_RAJ_burma_joint_war_coordination_trigger = {
	ENG_RAJ_burma_joint_political_trigger = yes
	ENG_RAJ_burma_joint_eng_war_leadership_trigger = yes
}
