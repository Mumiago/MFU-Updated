# Venezuela offers to 'protect' Curacao
country_event = {
	id = mtg_netherlands.1
	title = mtg_netherlands.1.t
	desc = mtg_netherlands.1.d
	picture = GFX_report_event_generic_sign_treaty3

	is_triggered_only = yes
	
	# Accept
	option = {
		name = mtg_netherlands.1.a
		effect_tooltip = {
			VEN = { transfer_state = 695 }
		}
		VEN = { country_event = { id = mtg_netherlands.2 days = 1 random_days = 3 } }
	}

	# They're just trying to get the islands for themselves...
	option = {
		name = mtg_netherlands.1.b
		effect_tooltip = {
			VEN = { add_state_claim = 695 }
		}
		VEN = { country_event = { id = mtg_netherlands.3 days = 1 random_days = 3 } }
		set_global_flag = ven_claims_abc_islands
	}

	# We can ill afford another crisis - give the British direct control
	option = {
		trigger = {
			AND = {
				is_in_faction_with = ENG
				NOT = { is_in_faction_with = USA }
			}
		}
		name = mtg_netherlands.1.c
		effect_tooltip = {
			ENG = { transfer_state = 695 }
		}
		ENG = { country_event = { id = mtg_netherlands.4 days = 1 random_days = 3 } }
	}

	# We can ill afford another crisis - give the Americans direct control
	option = {
		trigger = {
			is_in_faction_with = USA
		}
		name = mtg_netherlands.1.e
		effect_tooltip = {
			USA = { transfer_state = 695 }
		}
		USA = { country_event = { id = mtg_netherlands.4 days = 1 random_days = 3 } }
	}
}
