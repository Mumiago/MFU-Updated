# Treaty of Craiova - Romania Accepts
country_event = {
	id = bulgaria.1
	title = bulgaria.1.t
	desc = bulgaria.1.d
	picture = GFX_report_event_hitler_handshake
	
	is_triggered_only = yes
	
	option = {
		name = bulgaria.1.a
		
		if = {
			limit = {
				BUL = {
					exists = yes 
				}
			}
			BUL = { 
				transfer_state = 77
			}
		}
		if = {
			limit = {
				77 = { is_core_of = ROM }
				BUL = {
					exists = yes 
				}
			}
			ROM = {
				remove_state_core = 77
			}
		}
		hidden_effect = {
			news_event = { days = 1 id = news.21 }
		}
	}
}
