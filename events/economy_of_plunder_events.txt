country_event = {
	id = economy_of_plunder.27
	title = economy_of_plunder.27.t
	desc = economy_of_plunder.27.d
	picture = GFX_report_event_germany_politician_handshake
	
	is_triggered_only = yes
	
	option = { #Gladly we don't need that anyway# 
		name = economy_of_plunder.27.a 
		trigger = {
			has_equipment = {
				infantry_equipment < 1999
			}
			has_equipment = {
				support_equipment < 499
			}
			has_equipment = {
				horse_equipment_0 < 499
			}
		}
		GER = {
			send_equipment = {
				target = FROM 
				type = infantry_equipment
				amount = 2000
				old_prioritised = yes
			}
			send_equipment = {
				target = FROM 
				type = support_equipment
				amount = 500
				old_prioritised = yes
			}
			send_equipment = {
				target = FROM 
				type = horse_equipment_0
				amount = 500
				old_prioritised = yes
			}
		}
		FROM = {
			country_event = { id = economy_of_plunder.28 days = 5 }
		}
	}
	option = { #Re-route production lines# 
		name = economy_of_plunder.27.b 

		add_timed_idea = {
			idea = GER_production_assistance_to_RK
			days = 60 
		} 

		FROM = {
			add_equipment_to_stockpile = { type = infantry_equipment amount = 2000 }
			add_equipment_to_stockpile = { type = support_equipment amount = 500 }
			add_equipment_to_stockpile = { type = horse_equipment_0 amount = 500 }
		}
		FROM = {
			country_event = { id = economy_of_plunder.28 days = 5 }
		}
	}

	option = { #Impossible# 
		name = economy_of_plunder.27.c 
		every_state = {
			limit = {
				is_controlled_by = FROM 
			}
			add_resistance_target = {
				amount = 5
				tooltip = germany_didnt_help_us_tt
			}
		}
		FROM = {
			country_event = { id = economy_of_plunder.29 days = 5 }
		}
	}
}
